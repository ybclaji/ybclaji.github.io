# 基础环境

# 安装node

brew install node

# 安装webpack,全局

npm install webpack -g

# 安装vue-cli

npm install vue-cli -g

# 配置淘宝镜像

npm config set registry https://registry.npm.taobao.org

# 项目创建

# vue2.x创建,自建工程目录

vue init webpack badger-ui

# 启动测试

npm run dev
